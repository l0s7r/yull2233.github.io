javascript实现下拉菜单的显示与隐藏

js文件如下



    $(function(){
    $("#隐藏的对象名").hide();

    var wid_height = $(window).height();
    var sc_top = 0;
    $(window).scroll(function(){
        sc_top = $(this).scrollTop();
        if(sc_top>=wid_height)
            $("#隐藏的对象名").show();
        else
            $("#隐藏的对象名").hide();
    });

});
